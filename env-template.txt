# Firebase Configuration Template
# Kopiere diese Datei zu .env und fülle die Werte aus der Firebase Console ein

VITE_FIREBASE_API_KEY=your-api-key-here
VITE_FIREBASE_AUTH_DOMAIN=your-project.firebaseapp.com
VITE_FIREBASE_PROJECT_ID=your-project-id
VITE_FIREBASE_STORAGE_BUCKET=your-project.appspot.com
VITE_FIREBASE_MESSAGING_SENDER_ID=123456789
VITE_FIREBASE_APP_ID=1:123456789:web:abcdef123456

# Gemini API (optional für URL-Metadaten)
GEMINI_API_KEY=your-gemini-api-key-here

# Einfacher Passwortschutz (optional). Setze ein Passwort, um einen Prompt anzuzeigen.
VITE_APP_PASSWORD=

# Anleitung:
# 1. Gehe zu https://console.firebase.google.com/
# 2. Erstelle ein neues Projekt oder wähle ein bestehendes
# 3. Gehe zu Projekt-Einstellungen > Allgemein
# 4. Unter "Deine Apps" > Web-App > Konfiguration
# 5. Kopiere die Werte in diese .env Datei
